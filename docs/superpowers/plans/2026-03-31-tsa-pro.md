## Overview

TSA Pro is a single-file HTML app (~8,000-12,000 lines) implementing the world's first browser-based Trial Sequential Analysis. The file is structured with clearly marked sections:

```
<!-- === SECTION: HEAD + CSS === -->
<!-- === SECTION: HTML BODY === -->
<script>
// === DATA: BUILT-IN EXAMPLES ===
// === MATH ENGINE ===
// === BOUNDARY ENGINE ===
// === META-ANALYSIS ENGINE ===
// === TSA ENGINE ===
// === CHART ENGINE ===
// === UI ENGINE ===
// === INIT ===
<CLOSE_SCRIPT_TAG>   <!-- never literal </script> in code -->
```

**CRITICAL RULES:**
- NEVER write literal `</script>` inside a `<script>` block (including in comments). Use `${'<'}/script>` or string concatenation.
- Use `python` not `python3` on Windows.
- Every commit must leave the file as valid, loadable HTML.
- After structural edits, verify div balance: count of `<div` openings vs `</div>` closings.
- All `Math.random()` usage is forbidden for reproducible outputs -- use seeded PRNG if randomness is needed.
- `Blob` URLs must be revoked with `URL.revokeObjectURL()` after use.
- `|| fallback` drops zero -- use `?? fallback` for numeric fields.
- Never mix `??` with `||` or `&&` without parentheses.

```html
<script>
'use strict';

// === TAB NAVIGATION ===
(function initTabs() {
  const tabBtns = document.querySelectorAll('.tab-btn');
  const tabPanels = document.querySelectorAll('.tab-panel');

  function switchTab(tabName) {
    tabBtns.forEach(btn => {
      const isActive = btn.dataset.tab === tabName;
      btn.classList.toggle('active', isActive);
      btn.setAttribute('aria-selected', isActive ? 'true' : 'false');
    });
    tabPanels.forEach(panel => {
      panel.classList.toggle('active', panel.id === 'panel-' + tabName);
    });
  }

  tabBtns.forEach(btn => {
    btn.addEventListener('click', () => switchTab(btn.dataset.tab));
    btn.addEventListener('keydown', (e) => {
      const tabs = Array.from(tabBtns);
      const idx = tabs.indexOf(btn);
      let newIdx = -1;
      if (e.key === 'ArrowRight') newIdx = (idx + 1) % tabs.length;
      if (e.key === 'ArrowLeft') newIdx = (idx - 1 + tabs.length) % tabs.length;
      if (newIdx >= 0) {
        e.preventDefault();
        tabs[newIdx].focus();
        switchTab(tabs[newIdx].dataset.tab);
      }
    });
  });

  // Expose for programmatic use
  window.switchTab = switchTab;
})();

// === THEME TOGGLE ===
(function initTheme() {
  const html = document.documentElement;
  const toggle = document.getElementById('themeToggle');
  const icon = document.getElementById('themeIcon');
  const label = document.getElementById('themeLabel');

  // Restore saved preference
  const saved = localStorage.getItem('tsa-theme');
  if (saved) {
    html.setAttribute('data-theme', saved);
  }

  function updateUI() {
    const isDark = html.getAttribute('data-theme') === 'dark';
    icon.textContent = isDark ? '\u263E' : '\u2600';
    label.textContent = isDark ? 'Dark' : 'Light';
  }
  updateUI();

  toggle.addEventListener('click', () => {
    const current = html.getAttribute('data-theme');
    const next = current === 'dark' ? 'light' : 'dark';
    html.setAttribute('data-theme', next);
    localStorage.setItem('tsa-theme', next);
    updateUI();
  });
})();
```

```javascript
// === MATH ENGINE ===

/**
 * Normal CDF - Abramowitz & Stegun 26.2.17
 * P(Z <= x) for standard normal
 * Max error: 7.5e-8
 */
function normalCDF(x) {
  if (x === 0) return 0.5;
  const a1 = 0.254829592;
  const a2 = -0.284496736;
  const a3 = 1.421413741;
  const a4 = -1.453152027;
  const a5 = 1.061405429;
  const p = 0.3275911;

  const sign = x < 0 ? -1 : 1;
  const absX = Math.abs(x);
  const t = 1.0 / (1.0 + p * absX);
  const t2 = t * t;
  const t3 = t2 * t;
  const t4 = t3 * t;
  const t5 = t4 * t;
  const y = 1.0 - (a1 * t + a2 * t2 + a3 * t3 + a4 * t4 + a5 * t5) * Math.exp(-absX * absX / 2);

  return 0.5 * (1.0 + sign * y);
}

/**
 * Normal quantile (inverse CDF) - Beasley-Springer-Moro algorithm
 * Given p in (0,1), returns z such that P(Z <= z) = p
 */
function normalQuantile(p) {
  if (p <= 0) return -Infinity;
  if (p >= 1) return Infinity;
  if (p === 0.5) return 0;

  // Rational approximation for central region
  const a = [
    -3.969683028665376e+01,  2.209460984245205e+02,
    -2.759285104469687e+02,  1.383577518672690e+02,
    -3.066479806614716e+01,  2.506628277459239e+00
  ];
  const b = [
    -5.447609879822406e+01,  1.615858368580409e+02,
    -1.556989798598866e+02,  6.680131188771972e+01,
    -1.328068155288572e+01
  ];
  const c = [
    -7.784894002430293e-03, -3.223964580411365e-01,
    -2.400758277161838e+00, -2.549732539343734e+00,
     4.374664141464968e+00,  2.938163982698783e+00
  ];
  const d = [
     7.784695709041462e-03,  3.224671290700398e-01,
     2.445134137142996e+00,  3.754408661907416e+00
  ];

  const pLow = 0.02425;
  const pHigh = 1 - pLow;
  let q, r;

  if (p < pLow) {
    // Lower tail
    q = Math.sqrt(-2 * Math.log(p));
    return (((((c[0]*q + c[1])*q + c[2])*q + c[3])*q + c[4])*q + c[5]) /
           ((((d[0]*q + d[1])*q + d[2])*q + d[3])*q + 1);
  } else if (p <= pHigh) {
    // Central region
    q = p - 0.5;
    r = q * q;
    return (((((a[0]*r + a[1])*r + a[2])*r + a[3])*r + a[4])*r + a[5]) * q /
           (((((b[0]*r + b[1])*r + b[2])*r + b[3])*r + b[4])*r + 1);
  } else {
    // Upper tail
    q = Math.sqrt(-2 * Math.log(1 - p));
    return -(((((c[0]*q + c[1])*q + c[2])*q + c[3])*q + c[4])*q + c[5]) /
             ((((d[0]*q + d[1])*q + d[2])*q + d[3])*q + 1);
  }
}

/**
 * Compute Required Information Size (RIS) in total participants.
 *
 * @param {object} params
 *   Binary: { type:'binary', measure:'OR'|'RR'|'RD',
 *             anticipatedEffect, controlRate, alpha, power }
 *   Continuous: { type:'continuous', measure:'MD'|'SMD',
 *                 anticipatedEffect, sigma (for MD) or 1 (for SMD),
 *                 alpha, power }
 * @returns {number} RIS (total participants, both arms combined)
 */
function sampleSizeRIS(params) {
  const alpha = params.alpha ?? 0.05;
  const power = params.power ?? 0.80;
  const zAlpha = normalQuantile(1 - alpha / 2);
  const zBeta = normalQuantile(power);
  const zSum = zAlpha + zBeta;

  if (params.type === 'binary') {
    const pC = params.controlRate;
    const effect = params.anticipatedEffect;

    if (params.measure === 'RD') {
      // Risk Difference
      const pE = pC + effect;
      // Pooled variance approach
      const pBar = (pC + pE) / 2;
      return Math.ceil(4 * zSum * zSum * pBar * (1 - pBar) / (effect * effect));
    } else if (params.measure === 'RR') {
      // Relative Risk (log scale)
      const logRR = Math.log(Math.abs(effect));
      if (logRR === 0) return Infinity;
      return Math.ceil(4 * zSum * zSum / (logRR * logRR));
    } else {
      // OR (log scale)
      const logOR = Math.log(Math.abs(effect));
      if (logOR === 0) return Infinity;
      return Math.ceil(4 * zSum * zSum / (logOR * logOR));
    }
  } else {
    // Continuous: MD or SMD
    const delta = params.anticipatedEffect;
    const sigma = params.sigma ?? 1;
    if (delta === 0) return Infinity;
    return Math.ceil(2 * zSum * zSum * sigma * sigma / (delta * delta));
  }
}

/**
 * Compute diversity D^2 from heterogeneity I^2.
 * D^2 = I^2 / (1 - I^2) when I^2 < 1, else cap at large value.
 * @param {number} i2 - I-squared in [0, 1)
 * @returns {number} D-squared (diversity)
 */
function diversityAdjust(i2) {
  if (i2 == null || i2 < 0) return 0;
  if (i2 >= 1) return 100; // cap to avoid Infinity
  return i2 / (1 - i2);
}

/**
 * Alpha-spending function.
 * Maps information fraction t in [0,1] to cumulative alpha spent.
 *
 * Types:
 *   'obrien-fleming' : 2 - 2*Phi(z_{alpha/2} / sqrt(t))
 *   'pocock'         : alpha * ln(1 + (e-1)*t)
 *   'lan-demets-of'  : same as O'Brien-Fleming
 *   'lan-demets-pocock' : same as Pocock
 *
 * @param {number} t - information fraction in [0, 1]
 * @param {string} type - spending function type
 * @param {number} alpha - overall alpha level
 * @returns {number} cumulative alpha spent at fraction t
 */
function alphaSpending(t, type, alpha) {
  if (t <= 0) return 0;
  if (t >= 1) return alpha;
  alpha = alpha ?? 0.05;

  const zAlphaHalf = normalQuantile(1 - alpha / 2);

  switch (type) {
    case 'obrien-fleming':
    case 'lan-demets-of':
      return 2 - 2 * normalCDF(zAlphaHalf / Math.sqrt(t));
    case 'pocock':
    case 'lan-demets-pocock':
      return alpha * Math.log(1 + (Math.E - 1) * t);
    default:
      return alpha * t; // linear (fallback)
  }
}

/**
 * Beta-spending function.
 * Same functional forms as alpha-spending, but for type II error.
 *
 * @param {number} t - information fraction in [0, 1]
 * @param {string} type - spending function type
 * @param {number} beta - overall beta level (1 - power)
 * @returns {number} cumulative beta spent at fraction t
 */
function betaSpending(t, type, beta) {
  if (t <= 0) return 0;
  if (t >= 1) return beta;
  beta = beta ?? 0.20;

  const zBetaHalf = normalQuantile(1 - beta / 2);

  switch (type) {
    case 'obrien-fleming':
    case 'lan-demets-of':
      return 2 - 2 * normalCDF(zBetaHalf / Math.sqrt(t));
    case 'pocock':
    case 'lan-demets-pocock':
      return beta * Math.log(1 + (Math.E - 1) * t);
    default:
      return beta * t; // linear
  }
}
```

```javascript
// === BOUNDARY ENGINE ===

/**
 * Compute monitoring boundaries using recursive numerical integration.
 * Armitage-McPherson-Rowe algorithm with M grid points.
 *
 * @param {number[]} infoFractions - array of information fractions (ascending, last = IF_K)
 * @param {string} spendingType - spending function type
 * @param {number} alpha - overall alpha level (two-sided)
 * @param {number} [M=500] - number of grid points for integration
 * @returns {object[]} array of {t, z, alphaSpent} at each look
 */
function computeBoundaries(infoFractions, spendingType, alpha, M) {
  M = M ?? 500;
  const K = infoFractions.length;
  if (K === 0) return [];

  const results = [];
  const twoSided = true; // always two-sided in TSA Pro

  // For two-sided test, work with alpha/2 for each tail
  const halfAlpha = alpha / 2;

  // Cumulative alpha spent at each look (one-tail)
  const cumulativeSpent = [];
  for (let k = 0; k < K; k++) {
    cumulativeSpent.push(alphaSpending(infoFractions[k], spendingType, alpha) / 2);
  }

  // Incremental alpha at each look
  const incrementalAlpha = [];
  for (let k = 0; k < K; k++) {
    incrementalAlpha.push(k === 0 ? cumulativeSpent[0] : cumulativeSpent[k] - cumulativeSpent[k - 1]);
  }

  // Grid for numerical integration
  // We maintain a discrete density over the grid from look to look
  const gridMin = -8;
  const gridMax = 8;
  const dz = (gridMax - gridMin) / M;

  // Grid points (midpoints of intervals)
  const gridZ = [];
  for (let i = 0; i < M; i++) {
    gridZ.push(gridMin + (i + 0.5) * dz);
  }

  // Standard normal PDF
  function normalPDF(x) {
    return Math.exp(-0.5 * x * x) / Math.sqrt(2 * Math.PI);
  }

  // At look 1: standard normal density, find z1 from incremental alpha
  // P(Z > z1) = incrementalAlpha[0]
  // For first look, boundary is simply the quantile
  let z1 = normalQuantile(1 - incrementalAlpha[0]);
  results.push({
    t: infoFractions[0],
    z: z1,
    alphaSpent: cumulativeSpent[0] * 2 // two-sided total
  });

  if (K === 1) return results;

  // Initialize density on grid: standard normal truncated to [-z1, z1]
  let density = new Float64Array(M);
  for (let i = 0; i < M; i++) {
    if (gridZ[i] > -z1 && gridZ[i] < z1) {
      density[i] = normalPDF(gridZ[i]) * dz;
    }
  }

  // Recursive computation for looks 2..K
  for (let k = 1; k < K; k++) {
    const tPrev = infoFractions[k - 1];
    const tCurr = infoFractions[k];

    // Correlation between looks: rho = sqrt(t_{k-1} / t_k)
    const rho = Math.sqrt(tPrev / tCurr);
    const sdCond = Math.sqrt(1 - rho * rho); // conditional SD

    // Propagate density forward: Z_k | Z_{k-1} = z ~ N(rho*z, 1-rho^2)
    const newDensity = new Float64Array(M);
    for (let j = 0; j < M; j++) {
      // New grid point
      const zNew = gridZ[j];
      let sum = 0;
      for (let i = 0; i < M; i++) {
        if (density[i] <= 0) continue;
        // Conditional mean: rho * gridZ[i]
        const condMean = rho * gridZ[i];
        const condZ = (zNew - condMean) / sdCond;
        sum += density[i] * normalPDF(condZ) / sdCond * dz;
      }
      newDensity[j] = sum;
    }

    // Find z_k: the critical value such that P(|Z_k| > z_k AND not crossed before) = incrementalAlpha[k]
    // Bisection search over z_k
    let lo = 0;
    let hi = 8;
    for (let iter = 0; iter < 100; iter++) {
      const mid = (lo + hi) / 2;
      // Probability of crossing at this look: sum of density beyond +-mid
      let crossProb = 0;
      for (let j = 0; j < M; j++) {
        if (gridZ[j] > mid || gridZ[j] < -mid) {
          crossProb += newDensity[j];
        }
      }
      if (crossProb > incrementalAlpha[k]) {
        lo = mid; // boundary too small, increase
      } else {
        hi = mid; // boundary too large, decrease
      }
    }
    const zk = (lo + hi) / 2;
    results.push({
      t: infoFractions[k],
      z: zk,
      alphaSpent: cumulativeSpent[k] * 2
    });

    // Truncate density for next iteration (remove parts beyond boundary)
    density = new Float64Array(M);
    for (let j = 0; j < M; j++) {
      if (gridZ[j] > -zk && gridZ[j] < zk) {
        density[j] = newDensity[j];
      }
    }
  }

  return results;
}

/**
 * Compute futility boundaries using beta-spending.
 *
 * @param {number[]} infoFractions - array of information fractions
 * @param {string} spendingType - spending function type
 * @param {number} beta - overall beta (type II error = 1 - power)
 * @param {boolean} binding - true for binding futility, false for non-binding
 * @param {number} [M=500] - grid points
 * @returns {object[]} array of {t, z} at each look (inner futility bounds)
 */
function computeFutilityBoundaries(infoFractions, spendingType, beta, binding, M) {
  M = M ?? 500;
  const K = infoFractions.length;
  if (K === 0) return [];

  const results = [];

  // Cumulative beta spent at each look
  const cumulativeSpent = [];
  for (let k = 0; k < K; k++) {
    cumulativeSpent.push(betaSpending(infoFractions[k], spendingType, beta));
  }

  // Incremental beta
  const incrementalBeta = [];
  for (let k = 0; k < K; k++) {
    incrementalBeta.push(k === 0 ? cumulativeSpent[0] : cumulativeSpent[k] - cumulativeSpent[k - 1]);
  }

  // Grid setup
  const gridMin = -8;
  const gridMax = 8;
  const dz = (gridMax - gridMin) / M;
  const gridZ = [];
  for (let i = 0; i < M; i++) {
    gridZ.push(gridMin + (i + 0.5) * dz);
  }

  function normalPDF(x) {
    return Math.exp(-0.5 * x * x) / Math.sqrt(2 * Math.PI);
  }

  // Under H1 (true effect = drift), the Z-statistics follow N(drift_k, 1)
  // where drift_k = theta * sqrt(I_k) and at full information drift = z_alpha/2 + z_beta
  // For simplicity in the beta-spending framework, we compute boundaries under H0
  // and then the futility bound is the inner region where we would accept H0.

  // Non-binding futility: simple approach using quantiles of beta spending
  // Binding requires adjusting the efficacy boundary -- more complex

  // Simplified futility computation:
  // At each look k, the futility boundary is the z-value below which
  // we stop for futility. Using the quantile approach:
  // P(Z_k < f_k | H0, not yet stopped) = incrementalBeta[k]
  // Under H0, this is equivalent to finding f_k from the beta spending function

  if (!binding) {
    // Non-binding: futility boundaries are informational only
    // Use the standard quantile approach under the alternative
    // At look k, futility boundary ~ normalQuantile(cumulativeSpent[k])
    // This produces inner boundaries that get wider as information accumulates
    for (let k = 0; k < K; k++) {
      const betaCum = cumulativeSpent[k];
      // Under H1 with drift, the futility boundary is placed so that
      // the probability of crossing it (falsely accepting H0) = betaCum
      // Simplified: place boundary as a function of the beta spending
      // For O'Brien-Fleming beta spending, early boundaries are very negative (lenient)
      // For Pocock, they are more uniform
      const fk = normalQuantile(1 - betaCum);
      // Futility: inner boundary (we stop if |Z| < fk when fk > 0 is the inner boundary)
      // Convention: futility boundary is the z-value BELOW which we stop
      // In two-sided TSA, futility means Z is close to 0
      // Typical: f_k is positive and decreasing toward 0

      results.push({
        t: infoFractions[k],
        z: Math.max(fk, 0) // futility bound is always >= 0 (inner)
      });
    }
  } else {
    // Binding: more complex -- the futility boundary is part of the decision rule
    // and affects the efficacy boundary computation.
    // For now, use the same approach but mark as binding
    for (let k = 0; k < K; k++) {
      const betaCum = cumulativeSpent[k];
      const fk = normalQuantile(1 - betaCum);
      results.push({
        t: infoFractions[k],
        z: Math.max(fk, 0)
      });
    }
  }

  return results;
}
```

```javascript
// === META-ANALYSIS ENGINE ===

/**
 * Compute log Odds Ratio with 0.5 continuity correction.
 * @param {number} eE - events in experimental arm
 * @param {number} nE - total in experimental arm
 * @param {number} eC - events in control arm
 * @param {number} nC - total in control arm
 * @returns {{ yi: number, vi: number }} log(OR) and its variance
 */
function computeLogOR(eE, nE, eC, nC) {
  let aE = eE, bE = nE - eE, aC = eC, bC = nC - eC;

  // Continuity correction if any cell is zero
  if (aE === 0 || bE === 0 || aC === 0 || bC === 0) {
    aE += 0.5; bE = nE - eE + 0.5;
    aC += 0.5; bC = nC - eC + 0.5;
  }

  const yi = Math.log(aE * bC / (bE * aC));
  const vi = 1 / aE + 1 / bE + 1 / aC + 1 / bC;
  return { yi: yi, vi: vi };
}

/**
 * Compute log Risk Ratio with 0.5 continuity correction.
 * @param {number} eE - events in experimental arm
 * @param {number} nE - total in experimental arm
 * @param {number} eC - events in control arm
 * @param {number} nC - total in control arm
 * @returns {{ yi: number, vi: number }} log(RR) and its variance
 */
function computeLogRR(eE, nE, eC, nC) {
  let aE = eE, aC = eC;
  let adjNE = nE, adjNC = nC;

  // Continuity correction if any event count is zero
  if (aE === 0 || aC === 0) {
    aE += 0.5; aC += 0.5;
    adjNE += 0.5; adjNC += 0.5;
  }

  const yi = Math.log((aE / adjNE) / (aC / adjNC));
  const vi = (1 / aE - 1 / adjNE) + (1 / aC - 1 / adjNC);
  return { yi: yi, vi: vi };
}

/**
 * Compute Risk Difference.
 * @returns {{ yi: number, vi: number }}
 */
function computeRD(eE, nE, eC, nC) {
  const pE = eE / nE;
  const pC = eC / nC;
  const yi = pE - pC;
  const vi = pE * (1 - pE) / nE + pC * (1 - pC) / nC;
  return { yi: yi, vi: vi };
}

/**
 * Compute Mean Difference.
 * @param {number} mE - mean in experimental arm
 * @param {number} sdE - SD in experimental arm
 * @param {number} nE - N in experimental arm
 * @param {number} mC - mean in control arm
 * @param {number} sdC - SD in control arm
 * @param {number} nC - N in control arm
 * @returns {{ yi: number, vi: number }}
 */
function computeMD(mE, sdE, nE, mC, sdC, nC) {
  const yi = mE - mC;
  const vi = (sdE * sdE) / nE + (sdC * sdC) / nC;
  return { yi: yi, vi: vi };
}

/**
 * Compute Standardised Mean Difference (Hedges' g with small-sample correction).
 * @returns {{ yi: number, vi: number }}
 */
function computeSMD(mE, sdE, nE, mC, sdC, nC) {
  const df = nE + nC - 2;
  const pooledSD = Math.sqrt(((nE - 1) * sdE * sdE + (nC - 1) * sdC * sdC) / df);
  if (pooledSD === 0) return { yi: 0, vi: 0 };

  // Cohen's d
  const d = (mE - mC) / pooledSD;

  // Hedges' g (small-sample correction)
  const J = 1 - 3 / (4 * df - 1);
  const g = d * J;

  // Variance of g
  const vi = (nE + nC) / (nE * nC) + (g * g) / (2 * (nE + nC));
  return { yi: g, vi: vi };
}

/**
 * Compute effect size for a single study based on measure type.
 * @param {object} study - study data
 * @param {string} measure - 'OR', 'RR', 'RD', 'MD', 'SMD'
 * @returns {{ yi: number, vi: number }}
 */
function computeEffectSize(study, measure) {
  switch (measure) {
    case 'OR':
      return computeLogOR(study.eE, study.nE, study.eC, study.nC);
    case 'RR':
      return computeLogRR(study.eE, study.nE, study.eC, study.nC);
    case 'RD':
      return computeRD(study.eE, study.nE, study.eC, study.nC);
    case 'MD':
      return computeMD(study.mE, study.sdE, study.nE, study.mC, study.sdC, study.nC);
    case 'SMD':
      return computeSMD(study.mE, study.sdE, study.nE, study.mC, study.sdC, study.nC);
    default:
      throw new Error('Unknown measure: ' + measure);
  }
}

/**
 * Pool effects using REML random-effects model with HKSJ adjustment.
 * Fisher scoring for REML tau^2 estimation.
 *
 * @param {number[]} yi - effect sizes
 * @param {number[]} vi - variances
 * @returns {{ estimate: number, se: number, tau2: number, I2: number,
 *             ci_lower: number, ci_upper: number, z: number }}
 */
function poolREML(yi, vi) {
  const k = yi.length;
  if (k === 0) return null;

  if (k === 1) {
    const se = Math.sqrt(vi[0]);
    return {
      estimate: yi[0],
      se: se,
      tau2: 0,
      I2: 0,
      ci_lower: yi[0] - 1.96 * se,
      ci_upper: yi[0] + 1.96 * se,
      z: yi[0] / se
    };
  }

  // DerSimonian-Laird starting value for tau^2
  const wFE = vi.map(v => 1 / v);
  const sumW = wFE.reduce((a, b) => a + b, 0);
  const meanFE = wFE.reduce((s, w, i) => s + w * yi[i], 0) / sumW;
  const Q = wFE.reduce((s, w, i) => s + w * (yi[i] - meanFE) * (yi[i] - meanFE), 0);
  const sumW2 = wFE.reduce((s, w) => s + w * w, 0);
  const cVal = sumW - sumW2 / sumW;
  let tau2 = Math.max(0, (Q - (k - 1)) / cVal);

  // Fisher scoring for REML (max 50 iterations)
  for (let iter = 0; iter < 50; iter++) {
    const w = vi.map(v => 1 / (v + tau2));
    const sumWk = w.reduce((a, b) => a + b, 0);
    const muHat = w.reduce((s, wj, j) => s + wj * yi[j], 0) / sumWk;
    const sumW2k = w.reduce((s, wj) => s + wj * wj, 0);

    // Score (first derivative of REML log-likelihood with respect to tau2)
    let score = -0.5 * sumWk + 0.5 * sumW2k / sumWk;
    score += 0.5 * w.reduce((s, wj, j) => {
      const resid = yi[j] - muHat;
      return s + wj * wj * resid * resid;
    }, 0);

    // Fisher information (negative expected second derivative)
    const sumW3k = w.reduce((s, wj) => s + wj * wj * wj, 0);
    const fisherInfo = 0.5 * (w.reduce((s, wj) => s + wj * wj, 0)
      - 2 * sumW3k / sumWk + sumW2k * sumW2k / (sumWk * sumWk));

    if (fisherInfo <= 0) break;

    const tau2New = tau2 + score / fisherInfo;
    const delta = Math.abs(tau2New - tau2);
    tau2 = Math.max(0, tau2New);
    if (delta < 1e-10) break;
  }

  // Final pooled estimate with HKSJ adjustment
  const w = vi.map(v => 1 / (v + tau2));
  const sumWk = w.reduce((a, b) => a + b, 0);
  const estimate = w.reduce((s, wj, j) => s + wj * yi[j], 0) / sumWk;

  // HKSJ variance: q_HKSJ = (1/(k-1)) * sum(w_i * (y_i - mu)^2)
  const qHKSJ = w.reduce((s, wj, j) => {
    const resid = yi[j] - estimate;
    return s + wj * resid * resid;
  }, 0) / (k - 1);

  // HKSJ SE: multiply standard SE by sqrt(q_HKSJ)
  const seStandard = 1 / Math.sqrt(sumWk);
  const seHKSJ = seStandard * Math.sqrt(Math.max(1, qHKSJ));
  // The Math.max(1, qHKSJ) ensures HKSJ never narrows the CI vs standard

  // I^2
  const I2 = Math.max(0, Math.min(1, (Q - (k - 1)) / Math.max(Q, 1)));

  // Use t-distribution critical value for HKSJ (df = k-1)
  // Approximate t critical value using normal for now (conservative enough for k>=5)
  // Wilson-Hilferty for small k
  const dfT = k - 1;
  let tCrit;
  if (dfT >= 30) {
    tCrit = 1.96;
  } else {
    // Approximate t_{0.025, df} using an established formula
    // For small df, use lookup
    const tTable = {
      1: 12.706, 2: 4.303, 3: 3.182, 4: 2.776, 5: 2.571,
      6: 2.447, 7: 2.365, 8: 2.306, 9: 2.262, 10: 2.228,
      11: 2.201, 12: 2.179, 13: 2.160, 14: 2.145, 15: 2.131,
      16: 2.120, 17: 2.110, 18: 2.101, 19: 2.093, 20: 2.086,
      21: 2.080, 22: 2.074, 23: 2.069, 24: 2.064, 25: 2.060,
      26: 2.056, 27: 2.052, 28: 2.048, 29: 2.045
    };
    tCrit = tTable[dfT] ?? 1.96;
  }

  const ci_lower = estimate - tCrit * seHKSJ;
  const ci_upper = estimate + tCrit * seHKSJ;
  const z = estimate / seHKSJ;

  return {
    estimate: estimate,
    se: seHKSJ,
    tau2: tau2,
    I2: I2,
    ci_lower: ci_lower,
    ci_upper: ci_upper,
    z: z
  };
}

/**
 * Cumulative meta-analysis: pool studies 1..k for each k.
 *
 * @param {object[]} studies - array of study objects (sorted chronologically)
 * @param {string} measure - 'OR', 'RR', 'RD', 'MD', 'SMD'
 * @returns {object[]} array of cumulative results, one per look:
 *   { k, name, estimate, se, ci_lower, ci_upper, z, tau2, I2, N_cum, IF }
 *   (IF is set later by TSA engine once RIS is known)
 */
function cumulativeMA(studies, measure) {
  const results = [];
  const allYi = [];
  const allVi = [];
  let nCum = 0;

  for (let k = 0; k < studies.length; k++) {
    const study = studies[k];
    const es = computeEffectSize(study, measure);
    allYi.push(es.yi);
    allVi.push(es.vi);

    // Cumulative N
    if (study.nE != null && study.nC != null) {
      nCum += study.nE + study.nC;
    } else if (study.nE != null) {
      nCum += study.nE * 2; // continuous may only have nE/nC
    }

    // Pool all studies up to k
    const pooled = poolREML(allYi.slice(), allVi.slice());
    if (pooled === null) continue;

    results.push({
      k: k + 1,
      name: study.name,
      estimate: pooled.estimate,
      se: pooled.se,
      ci_lower: pooled.ci_lower,
      ci_upper: pooled.ci_upper,
      z: pooled.z,
      tau2: pooled.tau2,
      I2: pooled.I2,
      N_cum: nCum,
      IF: 0 // populated by TSA engine
    });
  }

  return results;
}
```

```javascript
/**

 *

 *   {

 *     type: 'binary'|'continuous',

 *     power: 0.80,

 *     futilityEnabled: false,
 *     futilityBinding: false,

 *     customD2: null (auto) or number
 *   }

 *   {

 *     boundaries: [...],         // efficacy boundary points {t, z}
 *     futility: [...] | null,    // futility boundary points or null
 *     ris: number,               // unadjusted RIS
 *     risAdj: number,            // diversity-adjusted RIS
 *     d2: number,                // D^2 diversity
 *     i2: number,                // I^2 from final cumulative pooling
 *     tau2: number,              // tau^2 from final pooling
 *     verdict: string,           // 'BENEFIT', 'HARM', 'FUTILITY', 'INCONCLUSIVE'
 *     verdictDetail: string,     // explanation text

function computeTSA(studies, settings) {
  const measure = settings.measure ?? 'OR';
  const alpha = settings.alpha ?? 0.05;
  const power = settings.power ?? 0.80;
  const beta = 1 - power;
  const spendingType = settings.alphaSpendingType ?? 'obrien-fleming';

  // Step 1: Cumulative meta-analysis
  const cumulative = cumulativeMA(studies, measure);
  if (cumulative.length === 0) {
    return {
      cumulative: [], boundaries: [], futility: null,
      ris: 0, risAdj: 0, d2: 0, i2: 0, tau2: 0,
      verdict: 'INCONCLUSIVE', verdictDetail: 'No studies provided.',
      crossedAtLook: null, adjustedCI: { lower: 0, upper: 0 },
      conditionalPower: 0, zCurve: []
    };
  }

  // Step 2: Get heterogeneity from final cumulative pooling
  const finalPool = cumulative[cumulative.length - 1];
  const i2 = finalPool.I2;
  const tau2 = finalPool.tau2;

  // D^2 (diversity): use custom if provided, otherwise compute from I^2

  // Step 3: Compute RIS

  if (anticipatedEffect == null) {
    // Auto-fill from pooled estimate
    if (measure === 'OR' || measure === 'RR') {
      anticipatedEffect = Math.exp(finalPool.estimate); // back-transform from log scale
    } else {
      anticipatedEffect = finalPool.estimate;
    }
  }

  if (controlRate == null && settings.type === 'binary') {
    // Compute average control event rate from studies

    studies.forEach(s => { totalEC += s.eC; totalNC += s.nC; });
    controlRate = totalEC / totalNC;

  const risParams = {
    type: settings.type ?? 'binary',
    measure: measure,
    anticipatedEffect: anticipatedEffect,
    controlRate: controlRate,
    sigma: settings.sigma ?? 1,
    alpha: alpha,
    power: power
  };
  const ris = sampleSizeRIS(risParams);
  const risAdj = Math.ceil(ris * (1 + d2));

  // Step 4: Compute information fractions
  const totalN = finalPool.N_cum;
  for (let k = 0; k < cumulative.length; k++) {
    cumulative[k].IF = cumulative[k].N_cum / risAdj;
  }

  // Step 5: Build the z-curve (information fraction, cumulative Z)
  const zCurve = cumulative.map(c => ({
    t: c.IF,
    z: c.z,
    k: c.k,
    name: c.name,
    N: c.N_cum
  }));

  // Step 6: Compute boundaries at the information fractions
  const infoFractions = cumulative.map(c => Math.min(c.IF, 2.0)); // cap at 200% for boundary calc
  // Only compute boundaries up to min(IF, 1.0) for the spending function
  // But we want boundary values at every look for plotting
  // The spending function naturally handles t > 1 (returns alpha)
  const boundaries = computeBoundaries(infoFractions, spendingType, alpha);

  // Step 7: Futility boundaries (if enabled)
  let futility = null;
  if (settings.futilityEnabled) {
    const betaType = settings.betaSpendingType ?? 'obrien-fleming';
    futility = computeFutilityBoundaries(
      infoFractions, betaType, beta, settings.futilityBinding ?? false
    );
  }

  // Step 8: Determine verdict
  let verdict = 'INCONCLUSIVE';
  let verdictDetail = '';
  let crossedAtLook = null;

  for (let k = 0; k < cumulative.length; k++) {
    const absZ = Math.abs(cumulative[k].z);
    const boundaryZ = boundaries[k] ? boundaries[k].z : Infinity;

    // Check efficacy crossing
    if (absZ >= boundaryZ) {
      crossedAtLook = k + 1;
      if (cumulative[k].z > 0) {
        verdict = 'HARM'; // positive Z = harm for protective interventions (depends on coding)
      } else {
        verdict = 'BENEFIT';
      }
      // For log(RR) < 0 or log(OR) < 0, negative Z means experimental is better
      // Adjust: if estimate < 0 (on log scale for OR/RR), it is a benefit
      if (measure === 'OR' || measure === 'RR') {
        if (cumulative[k].estimate < 0 && absZ >= boundaryZ) {
          verdict = 'BENEFIT';
        } else if (cumulative[k].estimate > 0 && absZ >= boundaryZ) {
          verdict = 'HARM';
        }
      } else if (measure === 'RD') {
        // For RD, negative = fewer events in experimental = benefit
        verdict = cumulative[k].estimate < 0 ? 'BENEFIT' : 'HARM';
      } else {
        // For MD/SMD, sign depends on direction. Assume positive = better for now.
        verdict = cumulative[k].estimate > 0 ? 'BENEFIT' : 'HARM';
      }
      break; // first crossing determines verdict
    }

    // Check futility crossing
    if (futility && futility[k]) {
      if (absZ <= futility[k].z) {
        crossedAtLook = k + 1;
        verdict = 'FUTILITY';
        break;
      }
    }
  }

  // Verdict detail text
  switch (verdict) {
    case 'BENEFIT':
      verdictDetail = 'The cumulative Z-curve crossed the monitoring boundary for benefit at look ' +
        crossedAtLook + ' (' + cumulative[crossedAtLook - 1].N_cum.toLocaleString() + ' participants, ' +
        (cumulative[crossedAtLook - 1].IF * 100).toFixed(1) + '% of RIS). ' +
        'The evidence is sufficient to conclude a beneficial effect.';
      break;
    case 'HARM':
      verdictDetail = 'The cumulative Z-curve crossed the monitoring boundary for harm at look ' +
        crossedAtLook + '. The evidence indicates a harmful effect of the intervention.';
      break;
    case 'FUTILITY':
      verdictDetail = 'The cumulative Z-curve crossed the futility boundary at look ' +
        crossedAtLook + '. The anticipated effect is unlikely to be demonstrated even with additional trials.';
      break;
    case 'INCONCLUSIVE':
      if (totalN >= risAdj) {
        verdictDetail = 'The accumulated evidence exceeds the required information size (' +
          totalN.toLocaleString() + ' of ' + risAdj.toLocaleString() + ' participants) ' +
          'but the Z-curve has not crossed the monitoring boundary. The result is inconclusive.';
      } else {
        verdictDetail = 'The accumulated evidence (' + totalN.toLocaleString() + ' of ' +
          risAdj.toLocaleString() + ' required participants, ' +
          (totalN / risAdj * 100).toFixed(1) + '% of RIS) has not reached the monitoring boundary. ' +
          'More trials are needed before a firm conclusion can be drawn.';
      }
      break;
  }

  // Step 9: TSA-adjusted confidence interval at final look
  const finalBoundary = boundaries[boundaries.length - 1];
  const adjustedCI = {
    lower: finalPool.estimate - (finalBoundary ? finalBoundary.z : 1.96) * finalPool.se,
    upper: finalPool.estimate + (finalBoundary ? finalBoundary.z : 1.96) * finalPool.se
  };

  // Step 10: Conditional power estimate
  // Under the current trend, what is the probability of crossing at the end?
  // Simplified: if Z at current IF, extrapolate to IF=1
  const currentIF = finalPool.IF;
  const currentZ = finalPool.z;
  let conditionalPower = 0;
  if (currentIF > 0 && currentIF < 1) {
    // Projected Z at full information: Z_proj = currentZ * sqrt(1 / currentIF)
    // But this overestimates. Better: drift parameter = Z / sqrt(IF)
    const drift = currentZ / Math.sqrt(currentIF);
    // Power = P(|Z_final| > z_boundary_final)
    // Under drift: Z_final ~ N(drift * sqrt(1), 1) = N(drift, 1)
    const finalBoundZ = boundaries[boundaries.length - 1] ? boundaries[boundaries.length - 1].z : 1.96;
    conditionalPower = 1 - normalCDF(finalBoundZ - Math.abs(drift)) +
                       normalCDF(-finalBoundZ - Math.abs(drift));
  } else if (currentIF >= 1) {
    conditionalPower = verdict !== 'INCONCLUSIVE' ? 1 : 0;
  }

  return {
    cumulative: cumulative,
    boundaries: boundaries,
    futility: futility,
    ris: ris,
    risAdj: risAdj,
    d2: d2,
    i2: i2,
    tau2: tau2,
    verdict: verdict,
    verdictDetail: verdictDetail,
    crossedAtLook: crossedAtLook,
    adjustedCI: adjustedCI,
    conditionalPower: Math.max(0, Math.min(1, conditionalPower)),
    zCurve: zCurve
  };
}
```

```javascript
/**
 * Draw the TSA monitoring diagram on canvas.
 * @param {HTMLCanvasElement} canvas
 * @param {object} tsaResult - output of computeTSA()
 * @param {object} opts - optional overrides
 */
function drawTSADiagram(canvas, tsaResult, opts) {
  opts = opts ?? {};
  var ctx = canvas.getContext('2d');
  var W = canvas.width;
  var H = canvas.height;
  var isDark = document.documentElement.getAttribute('data-theme') === 'dark';

  // Colors
  var bgColor = isDark ? '#0f0f1a' : '#f4f6f9';
  var gridColor = isDark ? '#2a2a4a' : '#d0d5dd';
  var textColor = isDark ? '#e0e0e8' : '#1a1a2e';
  var textMuted = isDark ? '#a0a0b8' : '#555570';
  var boundaryColor = '#ff5252';
  var futilityColor = '#00d4ff';
  var zCurveColor = isDark ? '#ffffff' : '#000000';
  var risLineColor = isDark ? '#555570' : '#999';
  var crossedColor = '#00e676';
  var notCrossedColor = '#ffc107';

  // Clear
  ctx.fillStyle = bgColor;
  ctx.fillRect(0, 0, W, H);

  // Margins
  var margin = { top: 50, right: 60, bottom: 70, left: 80 };
  var plotW = W - margin.left - margin.right;
  var plotH = H - margin.top - margin.bottom;

  // Determine axes ranges
  var maxIF = 1.0;
  tsaResult.zCurve.forEach(function(p) {
    if (p.t > maxIF) maxIF = p.t;
  });
  maxIF = Math.max(maxIF * 1.1, 1.05);

  var maxZ = 3;
  tsaResult.boundaries.forEach(function(b) {
    if (b.z > maxZ) maxZ = b.z;
  });
  tsaResult.zCurve.forEach(function(p) {
    if (Math.abs(p.z) > maxZ) maxZ = Math.abs(p.z);
  });
  maxZ = Math.ceil(maxZ + 0.5);

  // Scale functions
  function xScale(t) { return margin.left + (t / maxIF) * plotW; }
  function yScale(z) { return margin.top + plotH / 2 - (z / maxZ) * (plotH / 2); }

  // Grid lines
  ctx.strokeStyle = gridColor;
  ctx.lineWidth = 0.5;
  ctx.setLineDash([4, 4]);

  // Horizontal grid (Z values)
  for (var z = -maxZ; z <= maxZ; z++) {
    ctx.beginPath();
    ctx.moveTo(margin.left, yScale(z));
    ctx.lineTo(margin.left + plotW, yScale(z));
    ctx.stroke();
  }

  // Vertical grid (IF values)
  for (var t = 0; t <= maxIF; t += 0.2) {
    ctx.beginPath();
    ctx.moveTo(xScale(t), margin.top);
    ctx.lineTo(xScale(t), margin.top + plotH);
    ctx.stroke();
  }
  ctx.setLineDash([]);

  // Zero line
  ctx.strokeStyle = textMuted;
  ctx.lineWidth = 1;
  ctx.beginPath();
  ctx.moveTo(margin.left, yScale(0));
  ctx.lineTo(margin.left + plotW, yScale(0));
  ctx.stroke();

  // RIS vertical line at IF = 1.0
  ctx.strokeStyle = risLineColor;
  ctx.lineWidth = 1.5;
  ctx.setLineDash([6, 4]);
  ctx.beginPath();
  ctx.moveTo(xScale(1.0), margin.top);
  ctx.lineTo(xScale(1.0), margin.top + plotH);
  ctx.stroke();
  ctx.setLineDash([]);

  // Label RIS line
  ctx.fillStyle = textMuted;
  ctx.font = '11px ' + (isDark ? 'Consolas' : 'Segoe UI');
  ctx.textAlign = 'center';
  ctx.fillText('RIS', xScale(1.0), margin.top - 8);

  // Draw efficacy boundaries (upper)
  if (tsaResult.boundaries.length > 0) {
    ctx.strokeStyle = boundaryColor;
    ctx.lineWidth = 2.5;
    ctx.beginPath();
    tsaResult.boundaries.forEach(function(b, i) {
      var x = xScale(tsaResult.zCurve[i] ? tsaResult.zCurve[i].t : b.t);
      var y = yScale(b.z);
      if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
    });
    ctx.stroke();

    tsaResult.boundaries.forEach(function(b, i) {
      var x = xScale(tsaResult.zCurve[i] ? tsaResult.zCurve[i].t : b.t);
      var y = yScale(-b.z);
      if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
    });
    ctx.stroke();

    // Dots on boundary
    tsaResult.boundaries.forEach(function(b, i) {
      var x = xScale(tsaResult.zCurve[i] ? tsaResult.zCurve[i].t : b.t);
      ctx.fillStyle = boundaryColor;
      ctx.beginPath(); ctx.arc(x, yScale(b.z), 4, 0, 2 * Math.PI); ctx.fill();
      ctx.beginPath(); ctx.arc(x, yScale(-b.z), 4, 0, 2 * Math.PI); ctx.fill();
    });
  }

  // Draw futility boundaries (if present)
  if (tsaResult.futility && tsaResult.futility.length > 0) {
    ctx.strokeStyle = futilityColor;
    ctx.lineWidth = 2;
    ctx.setLineDash([6, 4]);

    ctx.beginPath();
    tsaResult.futility.forEach(function(f, i) {
      var x = xScale(tsaResult.zCurve[i] ? tsaResult.zCurve[i].t : f.t);
      var y = yScale(f.z);
      if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
    });
    ctx.stroke();
    // Lower mirror
    ctx.beginPath();
    tsaResult.futility.forEach(function(f, i) {
      var x = xScale(tsaResult.zCurve[i] ? tsaResult.zCurve[i].t : f.t);
      var y = yScale(-f.z);
      if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
    });
    ctx.stroke();
    ctx.setLineDash([]);
  }

  // Draw Z-curve
  if (tsaResult.zCurve.length > 0) {
    ctx.strokeStyle = zCurveColor;
    ctx.lineWidth = 2.5;

    tsaResult.zCurve.forEach(function(p, i) {
      var x = xScale(p.t);

      var isLast = (i === tsaResult.zCurve.length - 1);

      // Study labels (small, angled)
      if (tsaResult.zCurve.length <= 20) {
        ctx.save();
        ctx.fillStyle = textMuted;
        ctx.font = '9px ' + (isDark ? 'Consolas' : 'Segoe UI');

  ctx.rotate(-Math.PI / 2);
  ctx.fillText('Cumulative Z-statistic', 0, 0);

  ctx.fillText('Trial Sequential Analysis Monitoring Diagram', margin.left + plotW / 2, 25);

  ctx.font = '11px ' + (isDark ? 'Consolas' : 'Segoe UI');
  ctx.textAlign = 'center';
  for (var t = 0; t <= maxIF; t += 0.2) {

    ctx.fillStyle = textMuted;

  // Legend
  var legendX = margin.left + 10;
  var legendY = margin.top + 10;
  ctx.font = '11px ' + (isDark ? 'Consolas' : 'Segoe UI');
```
